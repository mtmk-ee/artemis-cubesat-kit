349  inline bool RequestExists(const std::string &request_name) const {
350 #if SIMPLEAGENT_IGNORE_CASE
351  // First check the argumented requests
352  for (auto request_pair : argumented_requests) {
353  if ( CompareAndIgnoreCase(request_name, request_pair.first) )
354  return true;
355  }
356  // Then check the non-argumented requests
357  for (auto request_pair : non_argumented_requests) {
358  if ( CompareAndIgnoreCase(request_name, request_pair.first) )
359  return true;
360  }
361  // At this point, the request is not stored
362  return false;
363 #else
364  // First check if the request exists as an argumented request
365  if ( argumented_requests.find(request_name) != argumented_requests.end() )
366  return true;
367  else
368  // Then check if the request exists as a non-argumented request
369  return non_argumented_requests.find(request_name) != non_argumented_requests.end();
370 #endif
371  }

383  template <typename DeviceType>
384  DeviceType* NewDevice(const std::string &name, bool crash_on_error = SIMPLEAGENT_STRICT_MODE) {
385 
386  // Check if a device with the given name already exists
387  if ( DeviceExists(name) ) {
388  printf("The device '%s' already exists\n", name.c_str());
389  throw DeviceAlreadyExistsException(name);
390  }
391  else {
392 
393  // Create the piece
394  int pindex = json_createpiece(this->cinfo, name, DeviceType::type);
395 
396  // Check if the piece was succesfully created
397  if ( pindex < 0 ) {
398 
399  printf("Failed to add %s device named '%s': %s\n", GetDeviceTypeString(DeviceType::type),
400  name.c_str(), cosmos_error_string(pindex).c_str());
401  if ( crash_on_error )
402  exit(pindex);
403 
404  return nullptr;
405  }
406 
407  // Get COSMOS indices
408  int cindex = this->cinfo->pieces[pindex].cidx;
409  int dindex = this->cinfo->device[cindex].all.didx;
410 
411  // Create the new device
412  DeviceType *device = new DeviceType(GetComplexAgent(), cindex, dindex);
413  device->SetName(name);
414  devices[name] = device;
415 
416  return device;
417  }
